\begin{frame}
  \frametitle{Abstract FSI problem on reference domain}

  Let $V = (\VF\times \QF)\times(\VS\times\QS)\times \VM$

  \vspace{0.5cm}

  Find $U = ((\UF, \PF), (\US, \PS), \DM) \in V$ such that
  \begin{equation*}
    R(U; v) = 0
  \end{equation*}
  for all $v = ((\vF, \qF), (\vS, \qS), \vM) \in
  \hat{V}$

  \begin{align*}
    R(U; v) &= \AF(U; (\vF, \qF)) +
               \AS(U; (\vS, \qS)) +
               \AAM(U; \vM)
  \end{align*}

\end{frame}
